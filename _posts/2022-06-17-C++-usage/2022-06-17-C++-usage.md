## Overload << cout
```cpp
#include <iostream>

class A {
public:
    int a;
    int b;
    A(int a_, int b_) {a = a_; b = b_;}
    friend std::ostream& operator<<(std::ostream& o, A& obj);
};

std::ostream& operator<<(std::ostream& o, A& obj) {
    o << obj.a << " " << obj.b << std::endl;
    return o;
}

int main() {
    A a(3, 4);
    std::cout << a; // 3 4
}
```
